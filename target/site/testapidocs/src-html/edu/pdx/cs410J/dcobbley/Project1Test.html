001package edu.pdx.cs410J.dcobbley;
002
003import org.junit.Test;
004import static org.junit.Assert.assertTrue;
005import edu.pdx.cs410J.InvokeMainTestCase;
006import static junit.framework.Assert.assertEquals;
007
008/**
009 * Tests the functionality in the {@link Project1} main class.
010 */
011public class Project1Test extends InvokeMainTestCase {
012
013    /**
014     * Invokes the main method of {@link Project1} with the given arguments.
015     */
016    private MainMethodResult invokeMain(String... args) {
017        return invokeMain( Project1.class, args );
018    }
019
020  /**
021   * Tests that invoking the main method with no arguments issues an error
022   */
023
024
025
026
027    @Test
028    public void testRegularCommandLineArguments(){
029        MainMethodResult result = invokeMain("david", "503-709-4866", "503-880-6960", "10/15/2015", "09:38", "10/15/2015", "09:42", "-print");
030        assertEquals(new Integer(0), result.getExitCode());
031        assertTrue(result.getOut().contains("david [Phone call from 503-709-4866 to 503-880-6960 from 10/15/2015 09:38 to 10/15/2015 09:42]"));
032    }
033
034    @Test
035    public void testAllCommandLineArguments(){
036        MainMethodResult result = invokeMain("david", "503-709-4866", "503-880-6960", "10/15/2015", "09:38", "10/15/2015", "09:42", "-print", "-README");
037        assertEquals(new Integer(0), result.getExitCode());
038        assertTrue(result.getOut().contains("Customer: david [Phone call from 503-709-4866 to 503-880-6960 from 10/15/2015 09:38 to 10/15/2015 09:42]\n" +
039                "README has been called\n" +
040                "This program is a phonebill application which takes a very specific amount of arguments\n" +
041                "You must provide a customer name, caller number, callee number, start time, and end time (mm/dd/yyyy mm:hh)\n" +
042                "\n" +
043                "Usage: java edu.pdx.cs410J.<login-id>.Project1 [options] <args>\n" +
044                "   args are (in this order):\n" +
045                "       customer               Person whose phone bill weâ€™re modeling\n" +
046                "       callerNumber           Phone number of caller\n" +
047                "       calleeNumber           Phone number of person who was called\n" +
048                "       startTime              Date and time call began (24-hour time)\n" +
049                "       endTime                Date and time call ended (24-hour time)\n" +
050                "   options are (options may appear in any order):\n" +
051                "       -print                 Prints a description of the new phone call\n" +
052                "       -README                Prints a README for this project and exits\n" +
053                "   Date and time should be in the format: mm/dd/yyyy hh:mm"));
054    }
055
056    @Test
057    public void NoCommandLineArgs(){
058        MainMethodResult result = invokeMain();
059        assertEquals(new Integer(1), result.getExitCode());
060        assertTrue(result.getOut().contains("Missing command line arguments"));
061    }
062
063
064}
